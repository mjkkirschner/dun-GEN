class TerrainBlock(object):
	def __init__(self, width, height, xpos, ypos, color):
		self.xpos = xpos
		self.ypos = ypos
		self.color = color
		self.width = width
		self.height = height
		x = width/2.0
		y = height/2.0
		
	

class WallBlock(TerrainBlock):
	def __init__(self, width, height, xpos, ypos, color):
		self.xpos = xpos
		self.ypos = ypos
		self.color = (0,0,0)
		self.width = width
		self.height = height
		x = width/2.0
		y = height/2.0
		

class FloorBlock(TerrainBlock):

		self.color = (.7,.5,.5)
		self.width = width
		self.height = height
		x = width/2.0
		y = height/2.0
